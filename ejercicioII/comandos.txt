# 1) Crear volumen
docker volume create db_data
docker volume ls

# 2) Ejecutar Postgres con variables, puerto y volumen
docker run -d \
  --name parcial-postgres \
  -e POSTGRES_USER=admin \
  -e POSTGRES_PASSWORD=12345 \
  -e POSTGRES_DB=parcial_db \
  -p 5432:5432 \
  -v db_data:/var/lib/postgresql/data \
  postgres:16

# 3) Entrar a PostgreSQL
docker exec -it parcial-postgres psql -U admin -d parcial_db

# 4) Validar persistencia
docker restart parcial-postgres
docker exec -it parcial-postgres psql -U admin -d parcial_db -c "SELECT * FROM estudiantes;"
